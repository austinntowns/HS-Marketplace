     ```typescript
     export async function triggerAlertMatching(listing: {
       id: string
       type: string
       city: string | null
       state: string | null
       askingPrice: number | null
       locationName: string | null
     }) {
       // Find all alerts where listing matches criteria
       // This is called from the listing approval action in Phase 2

       const allAlerts = await db
         .select({
           alert: alerts,
           user: users,
         })
         .from(alerts)
         .innerJoin(users, eq(alerts.userId, users.id))

       const matchingAlerts = allAlerts.filter(({ alert }) => {
         // State match: if alert has states, listing.state must be in the list
         if (alert.states && alert.states.length > 0) {
           if (!listing.state || !alert.states.includes(listing.state)) {
             return false
           }
         }

         // Type match: if alert has types, listing.type must be in the list
         if (alert.listingTypes && alert.listingTypes.length > 0) {
           if (!alert.listingTypes.includes(listing.type)) {
             return false
           }
         }

         // Price match
         if (alert.minPrice && listing.askingPrice && listing.askingPrice < alert.minPrice) {
           return false
         }
         if (alert.maxPrice && listing.askingPrice && listing.askingPrice > alert.maxPrice) {
           return false
         }

         return true
       })

       // Send emails to all matching alerts
       await Promise.all(
         matchingAlerts.map(({ alert, user }) =>
           sendAlertMatchEmail({
             buyerEmail: user.email!,
             buyerName: user.name || 'Hello Sugar Buyer',
             listingTitle: listing.locationName || `${listing.city}, ${listing.state}`,
             listingId: listing.id,
             listingType: listing.type,
             city: listing.city || '',
             state: listing.state || '',
             askingPrice: listing.askingPrice || 0,
           })
         )
       )

       return { matched: matchingAlerts.length }
     }
     ```
