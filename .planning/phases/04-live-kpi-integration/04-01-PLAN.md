    ```typescript
    import { z } from 'zod'

    // Individual month data point for trend charts
    export const kpiMonthSchema = z.object({
      month: z.string(),  // Format from API (e.g., "2025-10" or "Oct 2025")
      value: z.number(),
    })

    // Single KPI metric with last month value, MoM change, and trend
    export const kpiMetricSchema = z.object({
      lastMonth: z.number(),
      momChange: z.number(),  // Decimal: 0.12 = +12%
      trend: z.array(kpiMonthSchema),
      updatedAt: z.string(),  // ISO datetime
    })

    // Full KPI response - all fields optional for graceful degradation
    export const kpiResponseSchema = z.object({
      // Use EXACT field names from docs/kpi-api-contract.md
      revenue: kpiMetricSchema.optional(),
      newClients: kpiMetricSchema.optional(),
      bookings: kpiMetricSchema.optional(),
      membershipConversion: kpiMetricSchema.optional(),
    })

    export type KpiMonth = z.infer<typeof kpiMonthSchema>
    export type KpiMetric = z.infer<typeof kpiMetricSchema>
    export type KpiData = z.infer<typeof kpiResponseSchema>
    ```

    **CRITICAL:** The field names in this schema MUST match the actual API response field names documented in Task 2. If the API uses `new_clients` instead of `newClients`, use `new_clients` in the schema.

    // Generate 12 months of trend data ending at current month
    function generateTrend(baseValue: number, variance: number): Array<{ month: string; value: number }> {
      const months = []
      const now = new Date()
      for (let i = 11; i >= 0; i--) {
        const date = new Date(now.getFullYear(), now.getMonth() - i, 1)
        const month = date.toLocaleDateString('en-US', { month: 'short', year: 'numeric' })
        const value = Math.round(baseValue + (Math.random() - 0.5) * variance * 2)
        months.push({ month, value })
      }
      return months
    }
